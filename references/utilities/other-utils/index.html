        X (array-like): The input features.
        y (array-like): The target variable.
        model: The machine learning model to use for cross-validation.
        cv (int, optional): The number of folds for cross-validation. Default is 5.
    Returns:
        scores (array-like): The cross-validation scores.
    """

    def get_layers(units: pd.DataFrame) -> list:
    """
    Get the unique cortical layers from the given DataFrame.

    Parameters:
    units (pd.DataFrame): The DataFrame containing the units.

    Returns:
    list: A list of unique cortical layers.

    """
    # Get the unique layers
    layers = units['ecephys_structure_acronym'].unique()

    # Sort the layers
    layers.sort()

    # Return the layers
    return layers
